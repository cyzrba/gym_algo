#### `run()` 模板方法流程

```python
def run(self) -> None:
    args = self.parse_args()
    self.args = args
    self.validate_custom_args(args)
    args.output_dir.mkdir(parents=True, exist_ok=True)
    
    self.pose_model = YOLO(str(args.pose_model))
    input_views = self.resolve_input_views()
    self.validate_input_view_paths(input_views)
    
    view_results = {}
    for view_name, rgb_path, depth_path, ply_path in input_views:
        view_data = self.load_view_data(rgb_path, depth_path, ply_path)
        view_results[view_name] = self.process_view(view_name, view_data, args.output_dir)
    
    aggregated = self.aggregate_results(view_results)
    self.write_summary(aggregated)
    
    for view_name, view_result in view_results.items():
        self.print_view_result(view_name, view_result)
    self.print_summary(aggregated)
```

#### `parse_args()` 流程

```python
def parse_args(self) -> argparse.Namespace:
    parser = argparse.ArgumentParser(description=self.description)
    # 添加公共参数（input paths, rgb/depth format, pose model, output dir, depth window）
    self.add_common_args(parser)
    self.add_custom_args(parser)  # 子类扩展
    return parser.parse_args()
```

公共参数包括：`--rgb`, `--depth`, `--ply`, `--front-rgb`, `--front-depth`, `--front-ply`, `--back-rgb`, `--back-depth`, `--back-ply`, `--pose-model`, `--output-dir`, `--rgb-width`, `--rgb-height`, `--rgb-format`, `--depth-width`, `--depth-height`, `--depth-dtype`, `--depth-endian`, `--depth-window`
